#!/bin/bash

echo "Enter a number:"
read num

factorial=1
for ((i=1; i<=num; i++))
do
    factorial=$((factorial * i))
done

is_prime=1
if [ $num -lt 2 ]
then
    is_prime=0
else
    for ((i=2; i*i<=num; i++))
    do
        if [ $((num % i)) -eq 0 ]
        then
            is_prime=0
            break
        fi
    done
fi


{
    echo "Number: $num"
    echo "Factorial: $factorial"
    echo "Prime Number: $([ $is_prime -eq 1 ] && echo "Yes" || echo "No")"
} > result.txt

cat result.txt
